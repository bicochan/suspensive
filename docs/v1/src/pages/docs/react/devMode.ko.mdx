# devMode prop (experimental)

## Setup devMode prop

```jsx
import { Suspensive, SuspensiveProvider, SuspensiveDevTools } from '@suspensive/react'

const App = () => {
  const [suspensive] = useState(() => new Suspensive())

  return (
    // devMode work only with `<SuspensiveProvider/>`
    <SuspensiveProvider value={suspensive}>
      {/** children can use devMode now */}
      <SuspensiveDevTools />
    </SuspensiveProvider>
  )
}
```

### `<Suspense/>` devMode prop

```jsx /devMode/
import { Suspense } from '@suspensive/react'

const Example = () => (
  <Suspense
    fallback={<>loading...</>}
    devMode={{
      // devMode prop will work only in development mode
      showFallback: true,
    }}
  >
    children
  </Suspense>
)
```

### `<ErrorBoundary/>` devMode prop

```jsx /devMode/
import { ErrorBoundary } from '@suspensive/react'

const Example = () => (
  <ErrorBoundary
    fallback={({ error }) => <>{error.message}</>}
    devMode={{
      // devMode prop will work only in development mode
      showFallback: true,
    }}
  >
    children
  </ErrorBoundary>
)
```

```jsx /devMode/
import { ErrorBoundary } from '@suspensive/react'

const Example = () => (
  <ErrorBoundary
    fallback={({ error }) => <>{error.message}</>}
    devMode={{
      // devMode prop will work only in development mode
      showFallback: { after: 2000 },
    }}
  >
    children
  </ErrorBoundary>
)
```
